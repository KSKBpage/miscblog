Connect my Windows to DN42 network via a boarder router
```
netsh interface ipv4 set interface interface="Ethernet 4" dhcpstaticipcoexistence=enabled
netsh interface ipv4 add address "Ethernet 4" 172.22.77.32/28
route -p add 172.22.77.41/29 172.22.77.33
route -p add 172.22.77.41/29 172.22.77.33

route -p add 172.20.0.0/14 172.22.77.33
route -p add 172.31.0.0/16 172.22.77.33
route -p add 10.100.0.0/14 172.22.77.33
route -p add 10.127.0.0/16 172.22.77.33



netsh interface ipv6 set interface interface="Ethernet 4" dhcpstaticipcoexistence=enabled
netsh interface ipv6 add address "Ethernet 4" fd28:cb8f:4c92::/48
route -p add fd28:cb8f:4c92::/49 fd28:cb8f:4c92:1::1
route -p add fd28:cb8f:4c92:8000::/49 fd28:cb8f:4c92:1::1
route -p add fd00::/8 fd28:cb8f:4c92:1::1
```

On linux machine:
```
ip addr add 172.22.77.33/32 dev eth0
ip addr add fd28:cb8f:4c92:1::1/64 dev eth0
ip route add 172.22.77.32/32 dev eth0 src 172.22.77.33
ip route add fd28:cb8f:4c92::/64 dev eth0 src fd28:cb8f:4c92:1::1
```
